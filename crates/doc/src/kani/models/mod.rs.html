3//! Contains definitions that Kani compiler may use to model functions that are not suitable for
4//! verification or functions without a body, such as intrinsics.
5//!
6//! Note that these are models that Kani uses by default; thus, we keep them separate from stubs.
7
8// Definitions in this module are not meant to be visible to the end user, only the compiler.
9#[allow(dead_code)]
10mod intrinsics {
11    use std::fmt::Debug;
12    use std::mem::size_of;
13
14    /// Similar definition to portable SIMD.
15    /// We cannot reuse theirs since TRUE and FALSE defs are private.
16    /// We leave this private today, since this is not necessarily a final solution, so we don't
17    /// want users relying on this.
18    /// Our definitions are also a bit more permissive to comply with the platform intrinsics.
19    pub(super) trait MaskElement: PartialEq + Debug {
20        const TRUE: Self;
21        const FALSE: Self;
22    }
23
24    macro_rules! impl_element {
25        { $ty:ty } => {
26            impl MaskElement for $ty {
27                const TRUE: Self = -1;
28                const FALSE: Self = 0;
29            }
30        }
31    }
32
33    macro_rules! impl_unsigned_element {
34        { $ty:ty } => {
35            impl MaskElement for $ty {
36                // Note that in the declaration of the intrinsic it is documented that the lane
37                // values should be -1 or 0:
38                // <https://github.com/rust-lang/rust/blob/338cfd3/library/portable-simd/crates/core_simd/src/intrinsics.rs#L134-L144>
39                //
40                // However, MIRI and the Rust compiler seems to accept unsigned values and they
41                // use their binary representation. Thus, that's what we use for now.
42                /// All bits are 1 which represents TRUE.
43                const TRUE: Self = <$ty>::MAX;
44                /// All bits are 0 which represents FALSE.
45                const FALSE: Self = 0;
46            }
47        }
48    }
49
50    impl_element! { i8 }
51    impl_element! { i16 }
52    impl_element! { i32 }
53    impl_element! { i64 }
54    impl_element! { i128 }
55    impl_element! { isize }
56
57    impl_unsigned_element! { u8 }
58    impl_unsigned_element! { u16 }
59    impl_unsigned_element! { u32 }
60    impl_unsigned_element! { u64 }
61    impl_unsigned_element! { u128 }
62    impl_unsigned_element! { usize }

64    /// Calculate the minimum number of lanes to represent a mask
65    /// Logic similar to `bitmask_len` from `portable_simd`.
66    /// <https://github.com/rust-lang/portable-simd/blob/490b5cf/crates/core_simd/src/masks/to_bitmask.rs#L75-L79>
67    pub(super) const fn mask_len(len: usize) -> usize {
68        len.div_ceil(8)
69    }
70
71    #[cfg(target_endian = "little")]
72    unsafe fn simd_bitmask_impl<T, const LANES: usize>(input: &[T; LANES]) -> [u8; mask_len(LANES)]
73    where
74        T: MaskElement,
75    {
76        let mut mask_array = [0; mask_len(LANES)];
77        for lane in (0..input.len()).rev() {
78            let byte = lane / 8;
79            let mask = &mut mask_array[byte];
80            let shift_mask = *mask << 1;
81            *mask = if input[lane] == T::TRUE {
82                shift_mask | 0x1
83            } else {
84                assert_eq!(input[lane], T::FALSE, "Masks values should either be 0 or -1");
85                shift_mask
86            };
87        }
88        mask_array
89    }

91    /// Stub for simd_bitmask.
92    ///
93    /// It will reduce a simd vector (TxN), into an integer of size S (in bits), where S >= N.
94    /// Each bit of the output will represent a lane from the input. A lane value of all 0's will be
95    /// translated to 1b0, while all 1's will be translated to 1b1.
96    ///
97    /// In order to be able to do this pragmatically, we take additional parameters that are filled
98    /// by the compiler.
99    #[rustc_diagnostic_item = "KaniModelSimdBitmask"]
100    pub(super) unsafe fn simd_bitmask<T, U, E, const LANES: usize>(input: T) -> U
101    where
102        [u8; mask_len(LANES)]: Sized,
103        E: MaskElement,
104    {
105        // These checks are compiler sanity checks to ensure we are not doing anything invalid.
106        assert_eq!(
107            size_of::<U>(),
108            size_of::<[u8; mask_len(LANES)]>(),
109            "Expected size of return type and mask lanes to match",
110        );
111        assert_eq!(
112            size_of::<T>(),
113            size_of::<Simd::<E, LANES>>(),
114            "Expected size of input and lanes to match",
115        );
116
117        let data = &*(&input as *const T as *const [E; LANES]);
118        let mask = simd_bitmask_impl(data);
119        (&mask as *const [u8; mask_len(LANES)] as *const U).read()
120    }
121
122    /// Structure used for sanity check our parameters.
123    #[repr(simd)]
124    struct Simd<T, const LANES: usize>([T; LANES]);
125}

127#[cfg(test)]
128mod test {
129    use super::intrinsics as kani_intrinsic;
130    use std::intrinsics::simd::*;
131    use std::{fmt::Debug, simd::*};

133    /// Test that the `simd_bitmask` model is equivalent to the intrinsic for all true and all false
134    /// masks with lanes represented using i16.
135    #[test]
136    fn test_bitmask_i16() {
137        check_portable_bitmask::<_, i16, 16, u16>(mask16x16::splat(false));
138        check_portable_bitmask::<_, i16, 16, u16>(mask16x16::splat(true));
139    }
140
141    /// Tests that the model correctly fails if an invalid value is given.
142    #[test]
143    #[should_panic(expected = "Masks values should either be 0 or -1")]
144    fn test_invalid_bitmask() {
145        let invalid_mask = unsafe { mask32x16::from_int_unchecked(i32x16::splat(10)) };
146        assert_eq!(
147            unsafe { kani_intrinsic::simd_bitmask::<_, u16, i32, 16>(invalid_mask) },
148            u16::MAX
149        );
150    }
151
152    /// Tests that the model correctly fails if the size parameter of the mask doesn't match the
153    /// expected number of bytes in the representation.
154    #[test]
155    #[should_panic(expected = "Expected size of return type and mask lanes to match")]
156    fn test_invalid_generics() {
157        let mask = mask32x16::splat(false);
158        assert_eq!(unsafe { kani_intrinsic::simd_bitmask::<_, u16, i32, 2>(mask) }, u16::MAX);
159    }
160
161    /// Test that the `simd_bitmask` model is equivalent to the intrinsic for a few random values.
162    /// These values shouldn't be symmetric and ensure that we also handle endianness correctly.
163    #[test]
164    fn test_bitmask_i32() {
165        check_portable_bitmask::<_, i32, 8, u8>(mask32x8::from([
166            true, true, false, true, false, false, false, true,
167        ]));
168
169        check_portable_bitmask::<_, i32, 4, u8>(mask32x4::from([true, false, false, true]));
170    }
171
172    #[repr(simd)]
173    #[derive(Clone, Debug)]
174    struct CustomMask<T, const LANES: usize>([T; LANES]);
175
176    /// Check that the bitmask model can handle odd size SIMD arrays.
177    /// Since the portable_simd restricts the number of lanes, we have to use our own custom SIMD.
178    #[test]
179    fn test_bitmask_odd_lanes() {
180        check_bitmask::<_, [u8; 3], i128, 23>(CustomMask([0i128; 23]));
181        check_bitmask::<_, [u8; 9], i128, 70>(CustomMask([-1i128; 70]));
182    }
183
184    /// Compare the value returned by our model and the portable simd representation.
185    fn check_portable_bitmask<T, E, const LANES: usize, M>(mask: Mask<T, LANES>)
186    where
187        T: std::simd::MaskElement,
188        LaneCount<LANES>: SupportedLaneCount,
189        E: kani_intrinsic::MaskElement,
190        [u8; kani_intrinsic::mask_len(LANES)]: Sized,
191        u64: From<M>,
192    {
193        assert_eq!(
194            unsafe { u64::from(kani_intrinsic::simd_bitmask::<_, M, E, LANES>(mask)) },
195            mask.to_bitmask()
196        );
197    }
198
199    /// Compare the value returned by our model and the simd_bitmask intrinsic.
200    fn check_bitmask<T, U, E, const LANES: usize>(mask: T)
201    where
202        T: Clone,
203        U: PartialEq + Debug,
204        E: kani_intrinsic::MaskElement,
205        [u8; kani_intrinsic::mask_len(LANES)]: Sized,
206    {
207        assert_eq!(
208            unsafe { kani_intrinsic::simd_bitmask::<_, U, E, LANES>(mask.clone()) },
209            unsafe { simd_bitmask::<T, U>(mask) }
210        );
211    }

213    /// Similar to portable simd_harness.
214    #[test]
215    fn check_mask_harness() {
216        // From array doesn't work either. Manually build [false, true, false, true]
217        let mut mask = mask32x4::splat(false);
218        mask.set(1, true);
219        mask.set(3, true);
220        let bitmask = mask.to_bitmask();
221        assert_eq!(bitmask, 0b1010);
222
223        let kani_mask = unsafe { u64::from(kani_intrinsic::simd_bitmask::<_, u8, u32, 4>(mask)) };
224        assert_eq!(kani_mask, bitmask);
225    }
